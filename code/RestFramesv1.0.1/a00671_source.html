65  void RFsplinePDF::Initialize(){
66  Clear();
67 
68  int Ndata = GetNData();
69  m_Log << LogInfo << "Initializing with N = " << Ndata << " data points" <<LogEnd;
70 
71  int M = std::min(m_MMAX, std::max(3, Ndata/3));
72  int N = std::min(m_NMAX, M);
73 
74  double scale = double(M)/double(N);
75  double fac = 1.;
76 
77  std::vector<double> knots;
78 
79  int ndim = M;
80  int ndim_old;
81  fac *= m_r;
82 
83  double max_prob = 0;
84 
85  m_Log << LogInfo << "Initializing logsplines" << LogEnd;
86  for(int i = N; i >= 3; i--){
87  m_Log << LogInfo << "Initializing Ndim = " << ndim << LogEnd;
88  if(i == N){
89  m_lspl.push_back((LogsplinePDF*) new LogsplinePDF(GetVar(), ndim));
90  m_ndim.push_back(ndim);
91  m_off.push_back(false);
92  m_lspl[m_Nspline]->Initialize();
93  m_lspl[m_Nspline]->Optimize(GetData().GetWeightVar());
94  } else {
95  LogsplinePDF* ls1 = new LogsplinePDF(GetVar(), ndim);
96  LogsplinePDF* ls2 = new LogsplinePDF(GetVar(), ndim);
97  ls1->Initialize();
98  //ls2->Initialize();
99  ls2->Initialize(knots);
100  ls1->Optimize(GetData().GetWeightVar());
101  ls2->Optimize(GetData().GetWeightVar());
102 
103  bool bad1 = ls1->LogL() >= 0;
104  bool bad2 = ls2->LogL() >= 0;
105  if(bad1 && bad2){
106  delete ls1;
107  delete ls2;
108  m_Log << LogWarning << "BAAAD Ndim=" << ndim << LogEnd;
109  continue;
110  }
111 
112  if(bad1 || bad2){
113  m_Log << LogWarning << "BAAAD Ndim=" << ndim << " " << bad1 << " " << bad2 << " " << ls1->LogL() << LogEnd;
114  continue;
115  }
116 
117  if((ls1->LogL() > ls2->LogL() && !bad1) || bad2){
118  m_lspl.push_back(ls1);
119  m_ndim.push_back(ndim);
120  m_off.push_back(false);
121  delete ls2;
122  } else {
123  m_lspl.push_back(ls2);
124  m_ndim.push_back(ndim);
125  m_off.push_back(false);
126  delete ls1;
127  }
128  }
129 
130  double logl = m_lspl[m_Nspline]->LogL();
131  m_logl.push_back(logl);
132 
133  //double prob = logl - m_alpha*ndim - ndim*(ndim+1)/(Ndata-ndim-1);
134  double prob = logl - std::max(m_alpha, log(Ndata)/2.)*ndim/2.;
135  m_prob.push_back(prob);
136  if(i == N || prob > max_prob)
137  max_prob = prob;
138 
139  ndim_old = ndim;
140  ndim = double(i-1)*pow(scale, fac);
141  fac *= m_r;
142  knots = m_lspl[m_Nspline]->ReducedKnots(ndim_old-ndim);
143 
144  m_Nspline++;
145  }

149  void RFsplinePDF::Optimize(const RFDataVar& w, double beta){
150  int Ndata = GetNData();
151  double Wtot = 0.;
152  for(int d = 0; d < Ndata; d++)
153  Wtot += w[d];
154 
155  if(Wtot <= 0.)
156  return;
157 
158  double max_prob = 0.;
159  bool bmax = true;
160  for(int i = 0; i < m_Nspline; i++){
161  m_lspl[i]->Optimize(w, beta);
162  m_logl[i] = m_lspl[i]->LogL();
163  if(m_logl[i] >= 0. || std::isnan(m_logl[i]) || std::isinf(m_logl[i])){
164  m_off[i] = true;
165  m_prob[i] = 0.;
166  } else {
167  m_off[i] = false;
168  m_prob[i] = m_logl[i] - m_alpha*m_ndim[i]-m_ndim[i]*(m_ndim[i]+1)/(Ndata-m_ndim[i]-1);
169  //m_prob[i] = m_logl[i] - std::max(m_alpha, log(Wtot)/2.)*m_ndim[i];
170  if(bmax || m_prob[i] > max_prob){
171  max_prob = m_prob[i];
172  bmax = false;
173  }
174  }
175  }
176 
177  double prob_sum = 0;
178  for(int i = 0; i < m_Nspline; i++){
179  if(m_off[i]){
180  m_prob[i] = 0.;
181  } else {
182  m_prob[i] = exp(m_prob[i]-max_prob);
183  prob_sum += m_prob[i];
184  }
185  }
186 
187  double LogL = 0.;
188 
189  if(prob_sum > 0.){
190  for(int i = 0; i < m_Nspline; i++)
191  m_prob[i] /= prob_sum;
192 
193  for(int i = 0; i < Ndata; i++){
194  if(w[i] > 0.)
195  LogL += w[i]*log(Eval(GetVar()[i]));
196  }
197  }
198  SetLogL(LogL);
199  }
