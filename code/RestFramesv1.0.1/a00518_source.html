46  void Bspline::getonec1(int i, int j, std::vector<double>& c, const std::vector<double>& t){
47  c[i]=1.;
48  c[i+3]=(t[j+2]-t[j])*(t[j]-t[j+1])/((t[j+2]-t[j+3])*(t[j+1]-t[j+3]));
49  c[i+2]=(c[i+3]*(t[j+1]-t[j+3])+t[j+1]-t[j])/(t[j+2]-t[j+1]);
50  c[i+1]=-1.-c[i+3]-c[i+2];
51  }

53  void Bspline::InitC1(int i, const std::vector<double>& knots){
54  m_c1.clear();
55 
56  int k = knots.size();
57  int j;
58  std::vector<double> d;
59  for(j = 0; j < 4; j++)
60  d.push_back(0.);
61  double a, b, r;
62 
63 
64  for(j = 0; j < k+2; j++)
65  m_c1.push_back(0.);
66 
67  if(i == 0){
68  a = knots[2] - knots[0];
69  b = knots[2] - knots[1];
70  m_c1[2] = 1.;
71  m_c1[3] = -a/b;
72  m_c1[4] = -m_c1[2] - m_c1[3];
73  m_c1[1] = -3.*(knots[0]*knots[0] + m_c1[3]*knots[1]*knots[1] + m_c1[4]*knots[2]*knots[2]);
74  m_c1[0] = -knots[2]*m_c1[1] - m_c1[2]*a*a*a - m_c1[3]*b*b*b;
75  }
76  if(i == 1){
77  m_c1[k-1] = 1.;
78  m_c1[k] = (knots[k-3] - knots[k-1])/(knots[k-1] - knots[k-2]);
79  m_c1[k+1] = -m_c1[k] - m_c1[k-1];
80  }
81  if(i == 2)
82  getonec1(k-2, k-4, m_c1, knots);
83  if(i > 2){
84  getonec1(i-1, i-3, m_c1, knots);
85  getonec1(0, i-2, d, knots);
86  a=0.;
87  b=0.;
88  for(j = 0; j < 4; j++){
89  r = (knots[k-1] - knots[i+j-2]);
90  a += d[j]*r*r*r;
91  r = (knots[k-1] - knots[i+j-3]);
92  b += m_c1[i+j-1]*r*r*r;
93  }
94  for(j = 0; j < 4; j++)
95  m_c1[i+j] -= d[j]*b/a;
96  }
97  }
